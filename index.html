                const tempC = (formData.temperature - 32) * 5/9;

                if (tempC > 25) fluidPerHour += (tempC - 25) * 20;

                if (tempC < 10) fluidPerHour -= 100;

                

                // Humidity adjustment

                if (formData.humidity > 70) fluidPerHour += 100;

                if (formData.humidity < 40) fluidPerHour -= 50;
